    def extra=(extra)
      raise ArgumentError, "Extra must begin with \\301, \\302 or \\303" unless extra =~ /^[#{CODEA+CODEB+CODEC}]/n
      type = extra[/([#{CODEA+CODEB+CODEC}])/n, 1]
      data = extra[/[#{CODEA+CODEB+CODEC}](.*)/n, 1]
      @extra = class_for(type).new(data)
    end
